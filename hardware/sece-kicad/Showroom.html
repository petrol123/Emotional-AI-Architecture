<script>

window.onload = function() {

 

const waveCanvas = document.getElementById("waveCanvas");

const waveCtx = waveCanvas.getContext("2d");

 

const stateCanvas = document.getElementById("stateCanvas");

const stateCtx = stateCanvas.getContext("2d");

 

function resize() {

    waveCanvas.width = waveCanvas.clientWidth;

    waveCanvas.height = 250;

    stateCanvas.width = stateCanvas.clientWidth;

    stateCanvas.height = 250;

}

resize();

window.addEventListener("resize", resize);

 

let currentChannel = "ECG";

let t = 0;

 

document.querySelectorAll(".channel-btn").forEach(btn => {

    btn.addEventListener("click", function() {

        currentChannel = this.innerText;

    });

});

 

function generateSignal(x) {

    if (currentChannel === "ECG") return Math.sin((x+t)*0.05)*20;

    if (currentChannel === "RESP") return Math.sin((x+t)*0.01)*40;

    if (currentChannel === "EMG") return Math.sin((x+t)*0.3)*(Math.random()*25);

    if (currentChannel === "GSR") return Math.sin((x+t)*0.005)*60;

    if (currentChannel === "ACCEL") return Math.sin((x+t)*0.1)*30;

}

 

function drawWave() {

 

    waveCtx.clearRect(0,0,waveCanvas.width,waveCanvas.height);

    waveCtx.beginPath();

 

    let lastY = waveCanvas.height/2;

 

    for(let x=0;x<waveCanvas.width;x++) {

 

        let y = waveCanvas.height/2 + generateSignal(x);

 

        if(y < 10) {

            y = 10;

            document.getElementById("clampStatus").innerText = "YES";

            document.getElementById("clampStatus").className = "text-red-400";

        }

        else if(y > waveCanvas.height-10) {

            y = waveCanvas.height-10;

            document.getElementById("clampStatus").innerText = "YES";

            document.getElementById("clampStatus").className = "text-red-400";

        } else {

            document.getElementById("clampStatus").innerText = "NO";

            document.getElementById("clampStatus").className = "text-green-400";

        }

 

        if(x===0) waveCtx.moveTo(x,y);

        else waveCtx.lineTo(x,y);

 

        lastY = y;

    }

 

    waveCtx.strokeStyle = "#22d3ee";

    waveCtx.lineWidth = 2;

    waveCtx.stroke();

 

    let adc = Math.floor((lastY / waveCanvas.height) * 1023);

    document.getElementById("adcValue").innerText = adc;

 

    if(adc > 900) {

        document.getElementById("alertStatus").innerText = "ACTIVE";

        document.getElementById("alertStatus").className = "text-red-400";

    } else {

        document.getElementById("alertStatus").innerText = "INACTIVE";

        document.getElementById("alertStatus").className = "text-green-400";

    }

 

    t += 2;

}

 

function drawState(adc) {

 

    stateCtx.clearRect(0,0,stateCanvas.width,stateCanvas.height);

 

    stateCtx.strokeStyle="#334155";

    stateCtx.beginPath();

    stateCtx.moveTo(stateCanvas.width/2,0);

    stateCtx.lineTo(stateCanvas.width/2,stateCanvas.height);

    stateCtx.moveTo(0,stateCanvas.height/2);

    stateCtx.lineTo(stateCanvas.width,stateCanvas.height/2);

    stateCtx.stroke();

 

    let valence = (adc/1023 - 0.5) * stateCanvas.width;

    let arousal = Math.sin(t*0.02) * (stateCanvas.height/3);

 

    stateCtx.beginPath();

    stateCtx.arc(

        stateCanvas.width/2 + valence,

        stateCanvas.height/2 - arousal,

        8,0,Math.PI*2

    );

    stateCtx.fillStyle="#22d3ee";

    stateCtx.fill();

}

 

function animate() {

    drawWave();

    let adc = parseInt(document.getElementById("adcValue").innerText);

    drawState(adc);

    requestAnimationFrame(animate);

}

 

animate();

 

};

 

function describe(part) {

 

const descriptions = {

sensor: "Five biosignal acquisition channels (ECG, RESP, EMG, GSR, ACCEL). High impedance front-end.",

conditioning: "1M protection resistors, bias network, mid-rail reference for safe amplification.",

amp: "Differential amplification using rail-to-rail op-amps. Gain tailored per channel.",

filter: "Low-pass filtering for noise suppression and alias protection.",

clamp: "3.3V clamping stage to protect microcontroller ADC input.",

adc: "Arduino UNO Q ADC interface with hardware-level interrupt output."

};

 

document.getElementById("archDescription").innerText = descriptions[part];

}
